#!/bin/csh -f

#
# Program: data.omim.org

# Original Author: dbm
#
# Purpose:
#
#	Download the omim.txt file
#

cd `dirname $0` && source ./Configuration 

# Full URL the file
setenv URL_OMIM 'http://data.omim.org/downloads/7czWx_PqSx-QX7T4SYkfoQ/omim.txt.gz'

# Full path to download directory
setenv DOWNLOADS_DIR	${DATADOWNLOADS}/data.omim.org

# Full path to log file
setenv LOGFILE_OMIM	${MIRRORLOG}/data.omim.org.log

#
# wget
# -S = print the headers sent by HTTP servers and responses sent by FTP servers.
# -o = full path to log file
# -O = full path to output file
# -t = number of retries
# -r = reverse
# -k 7 = sort by the 7th column
# -nd = don't create directories
# -m = Turn on options suitable for mirroring. This option turns on recursion 
#    and time-stamping, sets infinite recursion depth and keeps FTP 
#    directory listings. It is currently equivalent to .-r -N -l 
#    inf --no-remove-listing.. 

# run mirror
mkdir -p ${DOWNLOADS_DIR}
cd ${DOWNLOADS_DIR}
wget -o ${LOGFILE_OMIM} -S -t 10 -nd -m "${URL_OMIM}"
